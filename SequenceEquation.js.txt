function permutationEquation(p) {
    const pos = {};
    
    
    for (let i = 0; i < p.length; i++) {
        pos[p[i]] = i + 1;
    }

    const result = [];

    for (let x = 1; x <= p.length; x++) {
        const y = pos[pos[x]];
        result.push(y);
    }

    return result;

}

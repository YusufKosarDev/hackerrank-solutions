function gridChallenge(grid) {
    
    let sortedGrid = grid.map(row => row.split('').sort().join(''));

    
    for (let col = 0; col < sortedGrid[0].length; col++) {
        for (let row = 1; row < sortedGrid.length; row++) {
            if (sortedGrid[row][col] < sortedGrid[row - 1][col]) {
                return "NO";
            }
        }
    }

    return "YES";

}

function gemstones(arr) {
    
    let commonChars = new Set(arr[0]);

    
    for (let i = 1; i < arr.length; i++) {
        let currentSet = new Set(arr[i]);
        
        
        commonChars = new Set(
            [...commonChars].filter(char => currentSet.has(char))
        );
    }

    return commonChars.size;

}

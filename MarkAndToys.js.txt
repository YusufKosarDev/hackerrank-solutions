function maximumToys(prices, k) {
    
    prices.sort((a, b) => a - b);

    let count = 0;
    let totalCost = 0;

    
    for (let i = 0; i < prices.length; i++) {
        
        if (totalCost + prices[i] <= k) {
            totalCost += prices[i];
            count++;
        } else {
            
            break;
        }
    }

    return count;

}

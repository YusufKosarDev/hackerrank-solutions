function miniMaxSum(arr) {
let max = arr[0];
let min = arr[0];
for (let i = 1; i < arr.length; i++) {
if (arr[i] > max) max = arr[i];
if (arr[i] < min) min = arr[i];
}
let total = 0;
for (let i = 0; i < arr.length; i++) {
    total += arr[i];
}
let minSum = total - max;
let maxSum = total - min;
console.log(minSum, maxSum);
}

function repeatedString(s, n) {
    let countInS = 0;
    for (let char of s) {
        if (char === 'a') {
            countInS++;
        }
    }

    const fullRepeats = Math.floor(n / s.length);

    
    let totalCount = fullRepeats * countInS;

    
    const remainder = n % s.length;
    for (let i = 0; i < remainder; i++) {
        if (s[i] === 'a') {
            totalCount++;
        }
    }

    return totalCount;

}

function timeConversion(s) {
let hour = Number(s.slice(0, 2));
    let minute = s.slice(3, 5);
    let second = s.slice(6, 8);
    let period = s.slice(8); // AM veya PM

    if (period === "AM") {
        if (hour === 12) {
            hour = 0;
        }
    } else { // PM
        if (hour !== 12) {
            hour += 12;
        }
    }

    hour = hour.toString().padStart(2, "0");

    return `${hour}:${minute}:${second}`;
}

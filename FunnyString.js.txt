function funnyString(s) {
    
    const r = s.split('').reverse().join('');
    
    for (let i = 1; i < s.length; i++) {
        const diff1 = Math.abs(s.charCodeAt(i) - s.charCodeAt(i - 1));
        const diff2 = Math.abs(r.charCodeAt(i) - r.charCodeAt(i - 1));
        
        if (diff1 !== diff2) {
            return "Not Funny";
        }
    }
    
    return "Funny";

}

function anagram(s) {
    
     if (s.length % 2 !== 0) return -1;

    let mid = s.length / 2;
    let left = s.slice(0, mid);
    let right = s.slice(mid);

    let freq = {};
    
    for (let char of left) {
        freq[char] = (freq[char] || 0) + 1;
    }

    let changes = 0;

    for (let char of right) {
        if (freq[char]) {
            freq[char]--;
        } else {
            changes++;
        }
    }

    return changes;

}

function bonAppetit(bill, k, b) {
    const totalBill = bill.reduce((acc, curr) => acc + curr, 0); 
    const annaActualTotal = totalBill - bill[k];
    
    const annaShouldPay = annaActualTotal / 2;
    
    if (b === annaShouldPay) {
        console.log("Bon Appetit");
    
} else {
    console.log(b - annaShouldPay);
}
}

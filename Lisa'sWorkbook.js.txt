function workbook(n, k, arr) {
    
    let page = 1;
    let specialCount = 0;

    for (let i = 0; i < n; i++) {
        let problems = arr[i];

        for (let problem = 1; problem <= problems; problem++) {
            
            if (problem === page) {
                specialCount++;
            }

            if (problem % k === 0 || problem === problems) {
                page++;
            }
        }
    }

    return specialCount;

}

function gameOfThrones(s) {
    
    const charCounts = {};

    
    for (let char of s) {
        charCounts[char] = (charCounts[char] || 0) + 1;
    }

    
    let oddCount = 0;
    for (let char in charCounts) {
        if (charCounts[char] % 2 !== 0) {
            oddCount++;
        }
    }

    
    return oddCount <= 1 ? "YES" : "NO";

}
